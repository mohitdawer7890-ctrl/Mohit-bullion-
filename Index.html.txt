    <script>
        // --- CONFIGURATION ---
        
        // Since real financial APIs cost money, we are running a SIMULATOR by default.
        // To use REAL data, you would need an API key from services like GoldAPI.io or MetalPriceAPI.
        
        // Base simulation prices (Approx India rates)
        let basePrice24k = 73500;
        let basePrice22k = 67400;
        let basePriceSilver = 87500;

        function updateRates() {
            // 1. SIMULATOR LOGIC (For Demo Purposes)
            // Randomly fluctuate prices by small amounts to look "Live"
            const randomFluctuation = (Math.random() - 0.5) * 50; 
            
            let current24k = basePrice24k + randomFluctuation;
            let current22k = basePrice22k + (randomFluctuation * 0.9); // 22k is cheaper
            let currentSilver = basePriceSilver + (randomFluctuation * 10);

            // 2. UPDATE DOM
            document.getElementById('price-24k').innerText = "₹" + Math.round(current24k).toLocaleString();
            document.getElementById('price-22k').innerText = "₹" + Math.round(current22k).toLocaleString();
            document.getElementById('price-silver').innerText = "₹" + Math.round(currentSilver).toLocaleString();

            // Update Time
            const now = new Date();
            document.getElementById('last-updated').innerText = now.toLocaleTimeString();

            // Determine Up/Down color (Random for simulator)
            updateArrows();
        }

        function updateArrows() {
            // Just random visual logic for the demo
            const metals = ['24k', '22k', 'silver'];
            metals.forEach(metal => {
                const el = document.getElementById(`change-${metal}`);
                const isUp = Math.random() > 0.5;
                el.innerHTML = isUp ? '▲ 0.5%' : '▼ 0.3%';
                el.className = isUp ? 'change up' : 'change down';
            });
        }

        // Run immediately
        updateRates();

        // Run every 5 seconds to simulate live feed
        setInterval(updateRates, 5000);
